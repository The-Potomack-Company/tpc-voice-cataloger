   ```typescript
   function groupByAssignee(sessions: SupabaseSession[], nameMap: Map<string, string>): { name: string; id: string; sessions: SupabaseSession[] }[] {
     const groups = new Map<string, SupabaseSession[]>()
     for (const session of sessions) {
       const key = session.assigned_to ?? 'unassigned'
       const group = groups.get(key) ?? []
       group.push(session)
       groups.set(key, group)
     }
     return Array.from(groups.entries())
       .map(([id, sessions]) => ({
         id,
         name: id === 'unassigned' ? 'Unassigned' : (nameMap.get(id) ?? id),
         sessions,
       }))
       .sort((a, b) => a.name.localeCompare(b.name))
   }
   ```
