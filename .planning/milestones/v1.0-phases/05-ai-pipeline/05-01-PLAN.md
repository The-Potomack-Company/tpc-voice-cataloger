       });

       export const catalogFieldsJsonSchema = zodToJsonSchema(catalogFieldsSchema);

    4. Create `proxy/wrangler.toml`:
       ```toml

       main = "src/index.ts"
       compatibility_date = "2025-01-01"

       # Set GEMINI_API_KEY via: npx wrangler secret put GEMINI_API_KEY
       ```

    5. Create `proxy/src/index.ts` — thin forwarder:
       - Define `interface Env { GEMINI_API_KEY: string; }`
       - Handle OPTIONS requests with CORS headers (Access-Control-Allow-Origin: *, Access-Control-Allow-Methods: POST, OPTIONS, Access-Control-Allow-Headers: Content-Type)
       - Handle POST requests:
         - Parse JSON body expecting `{ model: string, payload: object }`
         - Construct Gemini URL: `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${env.GEMINI_API_KEY}`
         - Forward payload as POST body with Content-Type: application/json
         - Return Gemini response with CORS headers
       - Return 405 for other methods
       - Wrap in try/catch, return 500 with `{ error: "Internal server error" }` on failure
       - Do NOT add auth, rate limiting, or business logic — this is a thin forwarder for a 2-5 person internal tool

    6. Run `cd proxy && npm install` to install wrangler and types

    NOTE: Do NOT run `wrangler deploy` — that requires user authentication. The proxy is ready for the user to deploy when they have their Gemini API key.
