   // JSON Import - Start
   if (request.action === TPC_CONSTANTS.MESSAGE_ACTIONS.START_IMPORT) {
     Logger.info('[ContentScript] START_IMPORT message received', {
       mode: request.data?.session?.mode,
       itemCount: request.data?.items?.length,
     });
     ImportController.startImport(request.data)
       .then(() => Logger.info('[ContentScript] ImportController.startImport() completed'))
       .catch(error => Logger.error('[ContentScript] ImportController.startImport() error:', error));
     sendResponse({ success: true });
     return true;
   }

   // JSON Import - Cancel
   if (request.action === TPC_CONSTANTS.MESSAGE_ACTIONS.CANCEL_IMPORT) {
     Logger.info('[ContentScript] CANCEL_IMPORT message received');
     ImportController.cancelImport();
     sendResponse({ success: true });
     return true;
   }
