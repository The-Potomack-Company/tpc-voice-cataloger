**1. Add `base64DataUrlToFile` helper function** (in PRIVATE HELPER FUNCTIONS section, after `setFieldIfPresent`):

```javascript
/**
 * Convert a base64 data URL (from TPC App export) to a File object for FileInjector.
 * @param {string} dataUrl - Full data URL: "data:image/jpeg;base64,/9j/..."
 * @param {string} filename - Desired filename (e.g., "photo-1.jpg")
 * @returns {File} File object ready for FileInjector.injectSingleFile()
 */
function base64DataUrlToFile(dataUrl, filename) {
  const [header, base64Data] = dataUrl.split(',');
  const mimeMatch = header.match(/data:(.*?);/);
  const mimeType = mimeMatch ? mimeMatch[1] : 'image/jpeg';
  const byteString = atob(base64Data);
  const ab = new ArrayBuffer(byteString.length);
  const ia = new Uint8Array(ab);
  for (let i = 0; i < byteString.length; i++) {
    ia[i] = byteString.charCodeAt(i);
  }
  return new File([ab], filename, { type: mimeType });
}
```

**4. Replace `processNextHouseItem` entirely** with a step-based state machine (per D-02, D-03, D-04):

```javascript
/**
 * Process the next item in house visit mode using a step-based state machine.
 * Steps: set-style -> fill-fields -> upload-photos -> save -> navigate
 * Page reloads expected after: set-style (if style changed), save, navigate.
 *
 * @param {string} step - Current step in the state machine
 * @param {number} photoIndex - Current photo index within the item (for upload-photos step)
 */
async function processNextHouseItem(step = 'set-style', photoIndex = 0) {
  if (isCancelled) return;

  // Check if all items processed
  if (step === 'set-style' && currentIndex >= importItems.length) {
    await completeImport();
    return;
  }

  const item = importItems[currentIndex];

  // ---- Step: set-style (per D-06) ----
  if (step === 'set-style') {
    const templateSelect = document.querySelector(TPC_CONSTANTS.RFC_SELECTORS.STYLE_FIELD);
    if (templateSelect && templateSelect.value !== TPC_CONSTANTS.RFC_SELECTORS.STYLE_GENERAL_VALUE) {
      Logger.info('ImportController: Setting Style to General', { currentValue: templateSelect.value });
      // Save state BEFORE the style change (triggers page reload via onchange -> jsSubmit)
      await saveImportState({ step: 'fill-fields', photoIndex: 0 });
      templateSelect.value = TPC_CONSTANTS.RFC_SELECTORS.STYLE_GENERAL_VALUE;
      templateSelect.dispatchEvent(new Event('change', { bubbles: true }));
      // Page will reload. checkAndResumeImport picks up with step: 'fill-fields'.
      return;
    }
    // Style already General or not found -- proceed to fill fields
    await processNextHouseItem('fill-fields', 0);
    return;
  }

  // ---- Step: fill-fields (per D-02, D-07) ----
  if (step === 'fill-fields') {
    Logger.info('ImportController: Filling fields for house item', { currentIndex, title: (item.title || '').substring(0, 50) });
    fillFieldsVerbatim(item);

    // Check if item has photos to upload
    const photos = item.photos || [];
    if (photos.length > 0) {
      await processNextHouseItem('upload-photos', 0);
    } else {
      // No photos -- skip to save (per D-02: text first, then photos, then save)
      await processNextHouseItem('save', 0);
    }
    return;
  }

  // ---- Step: upload-photos (per D-01) ----
  if (step === 'upload-photos') {
    const photos = item.photos || [];

    for (let pi = photoIndex; pi < photos.length; pi++) {
      if (isCancelled) return;

      const photo = photos[pi];
      const file = base64DataUrlToFile(photo.blob, `photo-${pi + 1}.jpg`);

      const fileInput = getFileInput();
      if (!fileInput) {
        Logger.error('ImportController: File input not found, skipping remaining photos');
        break;
      }

      // Optimistic state save BEFORE injection (per PortalUploadController pattern)
      await saveImportState({ step: 'upload-photos', photoIndex: pi + 1 });

      const injectResult = FileInjector.injectSingleFile(fileInput, file);
      if (!injectResult.success) {
        Logger.warn('ImportController: Photo injection failed, skipping', { pi, error: injectResult.error });
        continue;
      }

      const uploadResult = await UploadDetector.waitForUpload({
        timeout: TPC_CONSTANTS.PORTAL_UPLOAD_CONFIG.UPLOAD_TIMEOUT,
      });

      if (!uploadResult.success) {
        Logger.warn('ImportController: Photo upload timeout, skipping', { pi });
        continue;
      }

      Logger.info('ImportController: Photo uploaded', { pi, total: photos.length });

      // Delay between uploads (same timing as PortalUploadController)
      await new Promise(r => setTimeout(r, TPC_CONSTANTS.PORTAL_UPLOAD_CONFIG.PHOTO_UPLOAD_DELAY));
    }

    // All photos done (or skipped) -- proceed to save
    await processNextHouseItem('save', 0);
    return;
  }

  // ---- Step: save (per D-04) ----
  if (step === 'save') {
    updateStatistics('success');
    currentIndex++;
    updateProgressBar();

    // Check if this was the last item
    if (currentIndex >= importItems.length) {
      // Last item -- save and complete
      await saveImportState({ step: 'complete' });
      await NavigationHelper.clickSave();
      // After reload, completeImport will be triggered
      return;
    }

    // Save state BEFORE clicking Save (page will reload)
    await saveImportState({ step: 'navigate' });
    await NavigationHelper.clickSave();
    // Page reloads -> checkAndResumeImport picks up with step: 'navigate'
    return;
  }

  // ---- Step: navigate (per D-03) ----
  if (step === 'navigate') {
    Logger.info('ImportController: Navigating to next item', { currentIndex });

    // Try Next button first, fall back to Add (per D-03)
    const nextButton = document.querySelector(TPC_CONSTANTS.RFC_SELECTORS.NEXT_BUTTON);
    if (nextButton && !nextButton.disabled) {
      await saveImportState({ step: 'set-style' });
      nextButton.click();
      // Page reloads -> checkAndResumeImport picks up with step: 'set-style'
      return;
    }

    // Fall back to Add button
    const addButton = document.querySelector(TPC_CONSTANTS.RFC_SELECTORS.ADD_ITEM_BUTTON);
    if (addButton && !addButton.disabled) {
      Logger.info('ImportController: Next not available, using Add button');
      await saveImportState({ step: 'set-style' });
      addButton.click();
      // Page reloads -> checkAndResumeImport picks up with step: 'set-style'
      return;
    }

    Logger.error('ImportController: Neither Next nor Add button available');
    updateStatistics('error');
    await completeImport();
    return;
  }

  // ---- Step: complete (after last item save reload) ----
  if (step === 'complete') {
    await completeImport();
    return;
  }
}
```
